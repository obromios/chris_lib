def call
  chi2 = means.zip(mus, std_errs).map { |m, mu, se| (m.to_f - mu)**2 / se**2 }.sum
  p_value = PChiSquared.new.call(means.size, chi2)
  OpenStruct.new(pass?: p_value > threshold, p: p_value, chi2: chi2)
end
